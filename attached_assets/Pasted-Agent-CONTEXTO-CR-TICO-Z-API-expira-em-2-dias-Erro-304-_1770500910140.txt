     const globalAntiCacheMiddleware = (req, res, next) => {
       // Aplicar headers anti-cache em TODAS as respostas
       res.set({
         'Cache-Control': 'no-store, no-cache, must-revalidate, proxy-revalidate, max-age=0',
         'Pragma': 'no-cache',
         'Expires': '0',
         'Surrogate-Control': 'no-store',
         'ETag': 'W/"' + Date.now() + '"'
       });
       
       // Log detalhado
       const timestamp = new Date().toISOString();
       console.log(`[${timestamp}] ${req.method} ${req.path} - Anti-Cache Applied`);
       
       next();
     };
     
     module.exports = globalAntiCacheMiddleware;

11. Configuração de Upload (Multer):
    - Crie arquivo: backend/src/config/uploadConfig.js
    - Configuração:
      
      const multer = require('multer');
      const path = require('path');
      
      const storage = multer.diskStorage({
        destination: (req, file, cb) => {
          const uploadDir = path.join(__dirname, '../../uploads/branding');
          cb(null, uploadDir);
        },
        filename: (req, file, cb) => {
          const uniqueSuffix = Date.now() + '-' + Math.round(Math.random() * 1E9);
          cb(null, file.fieldname + '-' + uniqueSuffix + path.extname(file.originalname));
        }
      });
      
      const fileFilter = (req, file, cb) => {
        const allowedMimes = ['image/png', 'image/jpeg', 'image/gif', 'image/x-icon'];
        if (allowedMimes.includes(file.mimetype)) {
          cb(null, true);
        } else {
          cb(new Error('Tipo de arquivo não permitido'));
        }
      };
      
      const upload = multer({
        storage: storage,
        fileFilter: fileFilter,
        limits: { fileSize: 5 * 1024 * 1024 } // 5MB
      });
      
      module.exports = upload;

12. Processamento de Imagens (Sharp):

      const sharp = require('sharp');
      
      async function processLogo(filePath) {
        await sharp(filePath)
          .resize(500, 500, {
            fit: 'contain',
            background: { r: 255, g: 255, b: 255, alpha: 0 }
          })
          .png()
          .toFile(filePath);
      }
      
      async function processFavicon(filePath) {
        await sharp(filePath)
          .resize(128, 128, { fit: 'cover' })
          .png()
          .toFile(filePath);
      }
      
      module.exports = { processLogo, processFavicon };
